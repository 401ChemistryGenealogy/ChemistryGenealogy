    this.searchBarInputCheck = function(testExampleName){
        //Supply info to bar.
        this.mainSearchBar.sendKeys(testExampleName);

        //Check input
        expect(this.mainSearchBar.getText().toEqual(testExampleName));
    };

    this.testSearchNameOption = function(testExampleName){
        //Supply info to bar.
